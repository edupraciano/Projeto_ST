<script>
  var minhaUrl =
    "https://script.google.com/macros/s/AKfycbwxROeUFlV33V70esDgjRvHuxHzWP4lrBQyacP94AnJ3rFokHPHYXS3JmbH3XMPY5cYqA/exec";

  $(document).ready(function () {
    var user_name = localStorage.getItem('user_name');
    var password = localStorage.getItem('password');

    $('#close-btn').click(function () {
      $('#popup').hide();
    })

    if (user_name == "") {

      $('#div_login').fadeOut(0, function () {
        $('#div_login').fadeIn(550);
      });

      document.getElementById('div_login').style.display = "flex";

    } else {
      $.ajax({
        type: "POST",
        data: {
          qual_funcao: "autenticacao",
          user_name: user_name,
          password: password,
        },
        url: minhaUrl,
      }).done(function (data) {
        //console.log(data);
        if (data == "true") {
          // USUÁRIO LOGADO
          document.getElementById('div_login').style.display = 'none';
          document.getElementById('div_logout').style.display = "block";
          document.getElementById('menu_de_navegacao').style.display = 'block';
          document.getElementById('usuario_logado').innerHTML = 'Bem vindo(a), ' + user_name;
        } else {
          // USUÁRIO NÃO-LOGADO
          document.getElementById('div_login').style.display = "flex";
          //document.getElementById('div_logout').style.display = "none";
        }
      });
    };
  });

  function btn_login() {
    var user_name = document.getElementById("user_name").value;
    var password = document.getElementById("password").value;

    localStorage.setItem('user_name', '');
    localStorage.setItem('password', '');

    localStorage.setItem('user_name', user_name);
    localStorage.setItem('password', password);

    $.ajax({
      type: "POST",
      data: {
        qual_funcao: "autenticacao",
        user_name: user_name,
        password: password,
      },
      url: minhaUrl,
    }).done(function (data) {

      if (data == "true") {
        // USUÁRIO LOGADO
        console.log("Usuário Logado.");

        document.getElementById('div_logout').style.display = "block";
        document.getElementById('menu_de_navegacao').style.display = 'block';
        document.getElementById('div_login').style.display = "none";
        document.getElementById('usuario_logado').innerHTML = 'Bem vindo(a), ' + user_name;

        console.log("Usuário: " + user_name);
        console.log("Senha: " + password);

      } else {
        // USUÁRIO NÃO-LOGADO

        $('#div_login').effect('bounce', {
          times: 6,
          distance: 50
        }, 800);

        $('#popup').show();

        document.getElementById('div_login').style.display = "flex";
        //document.getElementById('div_logout').style.display = "none";
      };
    });
  }

  function btn_logout() {
    var user_name = document.getElementById("user_name").value;
    var password = document.getElementById("password").value;

    localStorage.setItem('user_name', '');
    localStorage.setItem('password', '');

    $('#div_login').fadeOut(0, function () {
      $('#div_login').fadeIn(550);
    })

    document.getElementById('div_login').style.display = "flex";
    document.getElementById('div_logout').style.display = "none";
    document.getElementById('menu_de_navegacao').style.display = 'none';
    document.getElementById('tabela_de_estabelecimentos').style.display = 'none';
    document.getElementById('formulario_de_cadastro_de_estabelecimento').style.display = 'none';

    try {
      $("#example").DataTable().clear();
      $("#example").DataTable().destroy();
    } catch (e) {
      console.log(e);
    }
    document.getElementById("user_name").value = "";
    document.getElementById("password").value = "";
  }

  function excluir_este_id() {
    var id_do_estabelecimento = document.getElementById('id_do_estabelecimento').value;

    ////////  PARTE DA AUTENTICAÇÃO DO USUÁRIO ////////////////       
    var user_name = localStorage.getItem('user_name');
    var password = localStorage.getItem('password');
    /////////////////////////////////////////////////////////

    if (id_do_estabelecimento == "") {
      alert('Selecione o Estabelecimento a ser Excluído!');
    } else {
      $.ajax({
        type: "POST",
        data: {
          qual_funcao: "remover_estabelecimento",
          id_do_estabelecimento: id_do_estabelecimento,
          user_name: user_name,
          password: password,
        },
        url: minhaUrl,
      }).done(function () {
        alert("Estabelecimento Excluído com Sucesso!");
        abre_tabela_de_estabelecimentos();
      });
    };
  }

  function abre_cadastro_de_estabelecimento() {
    document.getElementById("formulario_de_cadastro_de_estabelecimento").style.display = "block";
    document.getElementById("tabela_de_estabelecimentos").style.display = "none";
  }

  function abre_tabela_de_estabelecimentos() {
    document.getElementById("tabela_de_estabelecimentos").style.display = "block";
    document.getElementById("formulario_de_cadastro_de_estabelecimento").style.display = "none";
    lerDadosDaTabela();
  }

  function disparaAoClicar() {
    var id_do_estabelecimento = document.getElementById('id_do_estabelecimento').value;
    var nome_do_estabelecimento = document.getElementById('nome_do_estabelecimento').value;
    var produto = document.getElementById('produto').value;
    var chave = document.getElementById('chave').value;
    var maquina = document.getElementById('maquina').value;
    var endereco = document.getElementById('endereco').value;
    var responsavel = document.getElementById('responsavel').value;
    var contato = document.getElementById('contato').value;

    ///// PARTE DA AUTENTICAÇÃO DO USUÁRIO //////////////////
    var user_name = localStorage.getItem('user_name');
    var password = localStorage.getItem('password');
    /////////////////////////////////////////////////////////

    $.ajax({
      type: 'POST',
      data: {
        qual_funcao: 'entrada_de_estabelecimento',
        user_name: user_name,
        password: password,
        id_do_estabelecimento: id_do_estabelecimento,
        nome_do_estabelecimento: nome_do_estabelecimento,
        produto: produto,
        chave: chave,
        maquina: maquina,
        endereco: endereco,
        responsavel: responsavel,
        contato: contato,
      },
      url: minhaUrl,
    }).done(function (data) {
      document.getElementById('id_do_estabelecimento').value = '';
      document.getElementById('nome_do_estabelecimento').value = '';
      document.getElementById('produto').value = '';
      document.getElementById('chave').value = '';
      document.getElementById('maquina').value = '';
      document.getElementById('endereco').value = '';
      document.getElementById('responsavel').value = '';
      document.getElementById('contato').value = '';

      alert('Estabelecimento Cadastrado com Sucesso!');
      //alert(data);
      abre_tabela_de_estabelecimentos();
    });
  }

  function lerDadosDaTabela() {
    ////////  PARTE DA AUTENTICAÇÃO DO USUÁRIO ////////////////   
    var user_name = localStorage.getItem('user_name');
    var password = localStorage.getItem('password');
    /////////////////////////////////////////////////////////

    $.ajax({
      type: "POST",
      data: {
        user_name: user_name,
        password: password,
        qual_funcao: 'ler_estabelecimento',
      },
      url: minhaUrl,
    }).done(function (data) {

      try {
        $("#example").DataTable().clear();
        $("#example").DataTable().destroy();
      } catch (e) {
        console.log(e);
      }

      // GERAÇÃO DA TABELA
      var data = JSON.parse(data);

      new DataTable("#example", {
        data: data,
        deferRender: true,
        scrollCollapse: true,
        scroller: true,
        scrollY: 200,
      });

      // ADICIONANDO UM LISTINER
      var table = $('#example').DataTable();
      $('#example tbody').on('click', 'tr', function () {
        abre_cadastro_de_estabelecimento();
        var data = table.row(this).data();
        document.getElementById("id_do_estabelecimento").value = data[0];
        document.getElementById("nome_do_estabelecimento").value = data[1];
        document.getElementById("produto").value = data[2];
        document.getElementById("chave").value = data[3];
        document.getElementById("maquina").value = data[4];
        document.getElementById("endereco").value = data[5];
        document.getElementById("responsavel").value = data[6];
        document.getElementById("contato").value = data[7];
      });
    });
  }
